create procedure classify  @low decimal(5,2),@High decimal(5,2) as 
select SNAME,SDEPT,Grade from STUDENT s,GRADE g where s.SNO=g.SNO
and CNO in (select CNO from COURSE where CNAME='数据结构') and Grade>@low and Grade<@High
classify 70,99

/*group by*/
select SNAME,CNO,最高分 from GRADE g,
(select s.SNO, SNAME,MAX(Grade) 最高分 from 
STUDENT s,GRADE g where s.SNO=g.SNO group by s.SNO,SNAME) t
where t.SNO=g.SNO and t.最高分=g.Grade order by 最高分 desc;

过程化SQL

--条件控制
IF condition THEN
      Sequence_of_statements;
END IF;

IF condition THEN
     Sequence_of_statements;
ELSE 
     Sequence_of_statements; 
END IF;
--可以嵌套

--循环控制 提供EXIT、BREAK、LEAVE等结束循环
LOOP 
  Sequence_of_statements;
END LOOP

WHILE condition  LOOP
  Sequence_of_statements;
END LOOP;

FOR count IN  [REVERSE&#125; bound1..bound2 LOOP
   Sequence_of_statements;
END LOOP;
--将count设置为循环的下界bound1检查它是否小于上界bound2
--当指定了REVERSE时则将count设置为循环的上界bound2,检查count是否大与下界bound1
--越界退出循环,按照步长+1 -1，更新count

存储过程和函数

--存储过程定义,replace是为了替代重名的存储过程
create or replace proc 过程名 @参数1 数据类型...
AS <过程化SQL块(包括声明部分和可执行语句部分)>


--执行存储过程
call/perform 过程名 <和参数1类型相同的变量>...
--修改存储过程
alter proc 过程名1 rename to 过程名2;
--重新编译
alter proc 过程名 compile
--删除
drop proc 过程名;

--函数定义
 create or replace FUNCTION 函数名(参数1,..) RETURNS <类型>
 AS <过程化SQL块>;

--triggers
--删除的表用于存储 DELETE 和 UPDATE 语句所影响的行的副本。 在执行 DELETE 或 UPDATE 语句的过程中，行从触发器表中删除，并传输到删除的表中。 删除的表和触发器表通常没有相同的行。
--插入的表用于存储 INSERT 和 UPDATE 语句所影响的行的副本。 在执行插入或更新事务过程中，新行会同时添加到 inserted 表和触发器表中。 插入的表中的行是触发器表中的新行的副本
